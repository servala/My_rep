В Safari есть "Веб-инспектор". Опишите своими словами, что происходит в браузерах при клике на ссылку yandex.ru (полнота ответа – на ваше усмотрение). При нажатии на ссылку браузер запрашивает данные у сервера по средствам HTTP протокола, запрос является синхронным, т.е. браузер не может отправлять новый запрос, пока не получил ответ от предыдущего. Сам запрос состоит из нескольких этапов. Сначала идет поиск ближайщего DNS-сервера, где происходит перобразование имени сайта, написанное символами (URI), в его IP-дресс. По уникальному сетевому адрессу идет установка соединения с соответсвующим сервером. Далее браузер ожидает пока пакет данных дойдет до сервера, на сервере произойдет обработка запроса, и браузер получит ответ.

function exchange(){
   var multiply;
   switch( this.price.charAt( 0 ))
   {
       case '€':
           multiply = 40.725;
           break;
       case '¥':
           multiply = 0.415;
           break;
       default:
           multiply = 1;
   }
   return '₷' +
          parseInt( this.price.substring( 1 ) *
              multiply *  100 ) / 100;
}

function Car( manufacturer, model, year ) {
   this.manufacturer = manufacturer;
   this.model = model;
   this.year = !!year ? year : (new Date()).getFullYear();
   this.price = null;
}

Car.prototype.toString = function() {
   return this.getInfo();
}
Car.prototype.getInfo = function() {
  return [
      this.manufacturer,
      this.model,
      this.year
  ].join( ' ' );
}
Car.prototype.getDetailedInfo = function() {
   return 'Производитель: ' +
          this.manufacturer +
          '. Модель: ' +
          this.model +
          '. Год: ' +
          this.year;
}
Car.prototype.setPrice = function( price ) {
   this.price = price;
}
